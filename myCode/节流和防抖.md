## 节流／防抖

#### 总结
> 节流：每间隔一段时间执行一次 （eg:请求接口／输入框）
> 防抖：段时间快速出发，最后执行一次 （eg: 拉动窗口高度，在停止时执行）


```js

//防抖
//方法一：
var timer;
function debounce(handler, time) {
    clearTimeout(timer);
    timer = setTimeout(() => {
        handler()
    }, time)
}

function handler() {
    // to do...
    console.log(111)
}

window.onresize = function() {
    debounce(handler, 1000)
}

//方法二：
function throttle(method, context) {
    clearTimeout(method.tid)

    method.tid = setTimeout(() => {
        method.call(context);
    }, 1000)
}

function fn() {
    console.log(111)
}

window.onresize = function() {
    throttle(fn)
}

//节流

//方法一：
function throttle(fn,delay){
    let flag = true;
    return () => {
      if(!flag) return;
      flag = false;
      var timer = setTimeout(() => {
        fn();
        flag = true;
      }, delay)
    }
}

function fn() {
    console.log(111)
}

window.onresize = throttle(fn, 1000);

//方法二：
时间戳节流
```
